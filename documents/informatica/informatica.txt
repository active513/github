一.创建存放info相关信息的表空间
	CREATE TEMPORARY TABLESPACE BI_ETL_TMP  
	TEMPFILE 'E:\orcl\product\12.1.0\dbhome_1\oradata\bi_etl_data_space\info_data_tmp.dbf' 
	SIZE 50M  
	AUTOEXTEND ON  
	NEXT 50M MAXSIZE 20480M  
	EXTENT MANAGEMENT LOCAL;  
	--数据表空间
	CREATE TABLESPACE BI_ETL 
	DATAFILE 'E:\orcl\product\12.1.0\dbhome_1\oradata\bi_etl_data_space\info_data.dbf'         
	SIZE 500M                                                   
	AUTOEXTEND ON                                            
	NEXT 50M MAXSIZE 2048M                       
	EXTENT MANAGEMENT LOCAL;
	
二.创建域用户并分配权限，用来管理域
	--创建用户，并指定所用表空间，下同
	CREATE USER C##BI_DOMAIN 
	IDENTIFIED BY C##BI_DOMAIN 
	DEFAULT TABLESPACE BI_ETL 
	TEMPORARY TABLESPACE BI_ETL_TMP PROFILE DEFAULT ACCOUNT UNLOCK;
	--赋权
	GRANT CONNECT TO C##BI_DOMAIN;
	GRANT DBA TO C##BI_DOMAIN;
	GRANT RESOURCE TO C##BI_DOMAIN;
	GRANT UNLIMITED TABLESPACE TO C##BI_DOMAIN;
	GRANT SELECT ANY TABLE TO C##BI_DOMAIN;
	ALTER USER C##BI_DOMAIN DEFAULT ROLE ALL;
